第一个Gradle项目

有些目录及文件（如gradle.properties及subDemo模块）是后来加上去，src目录下的目录结构请参考Maven的标准目录结构

  • 理解Project及Task
  • 大概的书写过程及形式
  • 自定义一个任务
    如下：
//闭包
def createOther = {
    path - >
    File dir = new File(path)
    if (!dir.exists()) {
        dir.mkdirs()
    }
}
// 生成other目录及子目录
task makeOther() {
    def paths = ['src/main/other', 'src/main/other/WEB-INF']
    doFirst {
        paths.forEach(createOther)
    }
}
/**********************************/

//闭包
def createDir = {
    path - >
    File dir = new File(path)
    if (!dir.exists()) {
        dir.mkdirs()
    }
}
// 生成webapp目录及子目录
// 依赖于makeOther
task makeWeb() {
    dependsOn 'makeOther'
    def paths = ['src/main/webapp', 'src/main/webapp/WEB-INF']
    doFirst {
        paths.forEach(createDir)
    }
}

依赖管理

主要有两点：仓库管理及依赖引入

  • 仓库管理
repositories {
//    本地仓库
    mavenLocal()

//    中央仓库

//    私服

}
  • 依赖引入
dependencies {
//    两种坐标形式都可以
    testCompile group: 'junit', name: 'junit', version: '4.12'

}

版本冲突及解决方法

有两种方法来手动解决版本冲突(Groovy默认选取最高版本)：

  • 强制指定全局版本
configurations.all {
    resolutionStrategy {
//      关闭Gradle对版本冲突的默认处理（全部更改为最高版本）
        failOnVersionConflict()

//        指定版本
        force "org.hamcrest:hamcrest-core:1.3"
    }
}
  • 针对特定的Jar排除传递依赖
dependencies {
    compile ("junit:junit:4.12") {
        exclude group:"org.hamcrest", module:"hamcrest-core"
    }
}

发布

使用maven-publish插件进行发布

apply plugin: 'maven-publish'

//进行源码的打包任务
task sourceJar(type: Jar) {
    from sourceSets.main.allJava
}

//插件相关的配置
publishing {
    publications {
        maven(MavenPublication) {
            //指定group/artifact/version信息，可以不填。
            //默认使用项目group/name/version作为groupId/artifactId/version
            groupId project.group
            artifactId project.name
            version project.version
            //如果是war包填写components.web，如果是jar包填写components.java
            from components.java

            //配置上传源码
            artifact sourceJar {
                classifier "sources"
            }

        }
    }
    repositories {
        maven {
            //上传地址
            url = "xxx"
            //用户
            credentials {
                username 'kbrx93'
                password '123456'
            }
        }
    }
}
